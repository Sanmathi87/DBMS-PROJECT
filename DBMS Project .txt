5. SELF JOIN: Find pairs of cases handled by the same judge


SELECT c1.Case_ID AS Case1, c2.Case_ID AS Case2, j.Judge_Name FROM Case c1 JOIN Case c2 ON c1.Judge_ID = c2.Judge_ID AND c1.Case_ID < c2.Case_ID JOIN Judge j ON c1.Judge_ID = j.Judge_ID;
